async function sendData() { const input = document.getElementById("inputBox").value; if (!input) { alert("请输入内容！"); return; } try { const url = "https://script.google.com/macros/s/AKfycbxu-kBmw5Mj5LK_mjkoYhQKEPrr8burmZYp5czpRkVw2yofeRp3S_6pUxeNIEUrENLr/exec"; const response = await fetch(url, { method: "POST", headers: { "Content-Type": "application/json", // 确保使用 JSON 格式 "Origin": "https://player.okazu.cc", // 确保 Origin 与实际网站匹配 }, body: JSON.stringify({ name: input }), // 将数据作为 JSON 发送 }); const result = await response.json(); if (result.status === "success") { alert("成功记录到 Google 表格！"); document.getElementById("inputBox").value = ""; // 清空输入框 } else { alert("记录失败：" + result.message); } } catch (error) { alert("请求失败：" + error.message); } }
